```jsx
import { useState, useEffect } from "react";
function App() {
  const [toDo, setToDo] = useState("");
  const [toDos, setToDos] = useState([]);
  const onChange = (event) => {
    setToDo(event.target.value);
  };
  const onSubmit = (event) => {
    event.preventDefault();
    if (toDo === "") {
      return;
    }
    console.log(toDo);

    setToDo("");
    setToDos((currentArray) => {
      [toDo, ...currentArray]; // 새로운 형태의 array를 생성함.
    });
  };
  console.log(toDos);

  return (
    <div>
      <h1> To Dos ({toDos.length}) </h1>
      <form onSubmit={onSubmit}>
        <input
          onChange={onChange}
          value={toDo}
          type="text"
          placeholder="Write here your to do-line"
        />
        <button> Submit </button>
      </form>
    </div>
  );
}
export default App;
```

```jsx
import { useState, useEffect } from "react";
function App() {
  const [toDo, setToDo] = useState("");
  const [toDos, setToDos] = useState([]);
  const onChange = (event) => {
    setToDo(event.target.value);
  };
  const onSubmit = (event) => {
    event.preventDefault();
    if (toDo === "") {
      return;
    }
    setToDos((currentArray) => {
      return [toDo, ...currentArray];
    });
    setToDo("");
  };

  return (
    <div>
      <h1> To Dos ({toDos.length}) </h1>
      <form onSubmit={onSubmit}>
        <input
          onChange={onChange}
          value={toDo}
          type="text"
          placeholder="Write here your to do-line"
        />
        <button> Submit </button>
      </form>
      <hr />
      <ul>
        {toDos.map((item, index) => (
          <li key={index}>{item}</li>
        ))}
      </ul>
    </div>
  );
}
export default App;
```

```jsx
import { useState, useEffect } from "react";

function App() {
  const [loading, setLoading] = useState(true);
  const [coins, setCoins] = useState([]);
  const [trigger, setTrigger] = useState(0);
  const onClick = (event) => {
    setTrigger(trigger + 1);
  };
  useEffect(() => {
    fetch("https://api.coinpaprika.com/v1/tickers")
      .then((response) => {
        return response.json();
      })
      .then((json) => {
        console.log(json);
        setCoins(json);
        setLoading(false);
      });
  }, [trigger]);
  return (
    <div>
      <h1> The Coins!({coins.length})</h1>
      <button onClick={onClick}> Call Coins</button>
      <p> {loading ? "Loading..." : " Set on"}</p>
      <hr />
      <ul>
        {" "}
        {coins.map((coin, index) => {
          return (
            <li key={index}>
              {coin.name} ({coin.symbol}): <br />${coin.quotes.USD.price} USD
            </li>
          );
        })}
      </ul>
    </div>
  );
}
export default App;
```
